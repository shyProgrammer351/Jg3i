	import java.awt.*;

	import java.awt.event.*;

	import javax.swing.*;

	 

	// 질문 패널 

	class QuestionPanel extends JPanel

	{

		private Container frame;

		private CardLayout cards;

		private JRadioButton yes, no;

		private static int count = 0;

	 

		// 질문 패널 준비 

Q1b		public QuestionPanel( Container frame, CardLayout cards, String question ) {

			// 프레임에 다른 패널을 끼울 수 있도록 준비

			this.frame = frame;

			this.cards= cards;

	 

			// 질문과 답(예/아니오)을 제시 

			this.setLayout( new GridLayout( 3, 1 ) );

			this.add( new JLabel( question ) );

			yes = new JRadioButton( "예" );

			this.add( yes );

			no  = new JRadioButton( "아니오" );

			this.add( no );

			 

			// 답 선택시 반응할 리스너 등록 

			AnswerListener listener = new AnswerListener();

			yes.addActionListener( listener );

			no.addActionListener( listener );

Q1e		}

	 

		// 답 선택시 반응 

		private class AnswerListener implements ActionListener {

Q2b			public void actionPerformed( ActionEvent event ) {

				if ( yes.isSelected() )

					 count++;

				// 5개 질문 중 '예'가 4개 이상이면 내향인 출력 

				if ( count >= 4 )

					cards.last( frame );

				// 아니면 질문 계속하기 

				else

					cards.next( frame );

			}

Q2e		}

	}
